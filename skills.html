# Skills Assessment Protocol
def evaluate_skills():
    technologies = {
        'Python': {'level': 4, 'years': 1.5},
        'FastAPI': {'level': 4, 'years': 1},
        'JavaScript': {'level': 3, 'years': 1},
        'React': {'level': 2, 'years': 0.5},
        'Supabase': {'level': 3, 'years': 0.8},
        'Git': {'level': 3, 'years': 1},
        'Tailwind': {'level': 3, 'years': 0.7}
    }
    
    for tech, details in technologies.items():
        stars = '★' * details['level'] + '☆' * (5 - details['level'])
        print(f"▪ {tech} → {stars}")
        
    return "Skills assessment complete"

evaluate_skills()
