# Project Archive System
class ProjectManager:
    def __init__(self):
        self.completed_missions = []
        self.abandoned_missions = []
        
    def add_completed_project(self, name, tech_stack, description):
        project = {
            'name': name,
            'technologies': tech_stack,
            'description': description,
            'status': 'COMPLETED',
            'impact': 'Positive'
        }
        self.completed_missions.append(project)
        
    def archive_project(self, name, reason):
        project = {
            'name': name,
            'status': 'ARCHIVED',
            'reason': reason,
            'lessons_learned': 'Experience gained'
        }
        self.abandoned_missions.append(project)
        
portfolio = ProjectManager()
